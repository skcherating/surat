        function updateLetter() {

            // Get values from form

            const dateVal = document.getElementById('dateInput').value;

            const parentVal = document.getElementById('parentInput').value;

            const studentVal = document.getElementById('studentInput').value;

            const classVal = document.getElementById('classInput').value;

            const issueVal = document.getElementById('issueInput').value;


            // Format date (optional, making it look nicer like DD-MM-YYYY)

            let formattedDate = "[Tarikh]";

            if(dateVal) {

                const dateObj = new Date(dateVal);

                const day = String(dateObj.getDate()).padStart(2, '0');

                const month = String(dateObj.getMonth() + 1).padStart(2, '0'); // Months are 0-11

                const year = dateObj.getFullYear();

                formattedDate = `${day}/${month}/${year}`;

            }


            // Update letter text

            document.getElementById('outDate').innerText = dateVal ? formattedDate : "[Tarikh]";

            document.getElementById('outParent').innerText = parentVal || "[Nama Ibu Bapa / Penjaga]";

            

            document.getElementById('outStudent').innerText = studentVal || "[Nama Murid]";

            document.getElementById('outStudentBox').innerText = studentVal || "[Nama Murid]";

            

            document.getElementById('outClass').innerText = classVal || "[Kelas]";

            document.getElementById('outClassBox').innerText = classVal || "[Kelas]";

            

            document.getElementById('outIssue').innerText = issueVal || "[Sila nyatakan isu disiplin di ruang borang sebelah kiri]";

        }

    </script>
